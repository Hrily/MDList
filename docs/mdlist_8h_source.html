317 void MDList<T>::insert (ULL key, T val)
318 {
319  if (key < 0 || key >= N)
320  return;
321  start:
322  vector<int> coordinates = keyToCoordinates(key, this->D, this->N);
323  pair<Node<T>*, Node<T>*> p = locatePredecessor(coordinates);
324  Node<T>* predecessor = p.first;
325  Node<T>* current = p.second;
326  // Lock for thread safety
327  if (predecessor != NULL && !predecessor->try_lock())
328  goto start;
329  if (current != NULL && !current->try_lock())
330  {
331  if (predecessor != NULL)
332  predecessor->unlock();
333  goto start;
334  }
335  // Check predecessor and current are still valid
336  p = locatePredecessor(coordinates);
337  if (predecessor != p.first || current != p.second)
338  {
339  if (predecessor != NULL)
340  predecessor->unlock();
341  if (current != NULL)
342  current->unlock();
343  goto start;
344  }
345  // Check if key already exists.
346  if (current != NULL && key == current->getKey())
347  {
348  // Update value and return.
349  current->setValue(val);
350  if (predecessor != NULL)
351  predecessor->unlock();
352  current->unlock();
353  return;
354  }
355  // Key doesn't exits, create new Node.
356  Node<T>* node = new Node<T>(key, this->D, this->N, val);
357  // Find position of node in predecessor
358  int d = 0;
359  while (d < this->D && coordinates[d] <= predecessor->getCoordinates()[d])
360  d++;
361  if (d >= this->D)
362  throw "Given key is out of key space";
363  int _d = d;
364  // If current is NULL, then node is _dth child of predecessor.
365  if (current == NULL)
366  {
367  predecessor->setChild(_d, node);
368  predecessor->unlock();
369  return;
370  }
371  // Assign appropriate childrens to node.
372  while (d < this->D)
373  {
374  // Check if current can be dth child of node.
375  if (node->getCoordinates()[d] < current->getCoordinates()[d]){
376  node->setChild(d, current);
377  break;
378  }
379  // Else dth child of current is dth child of node
380  node->setChild(d, current->getChild(d));
381  current->setChild(d, NULL);
382  d++;
383  }
